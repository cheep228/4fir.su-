local function disconnectAssistHitFeedback()
    if assistHitFeedbackConn then
        assistHitFeedbackConn:Disconnect()
        assistHitFeedbackConn = nil
    end

    assistHitFeedbackSource = nil
end

local function getAssistHitRemotes()
    local remotesFolder = ReplicatedStorage:FindFirstChild("GameRemotes")
    if not remotesFolder then
        disconnectAssistHitFeedback()
        return nil, nil
    end

    local tryAttackRemote = remotesFolder:FindFirstChild("TryAttackTarget")
    local feedbackRemote = remotesFolder:FindFirstChild("AssistHitFeedback")

    if feedbackRemote ~= assistHitFeedbackSource then
        disconnectAssistHitFeedback()

        if feedbackRemote then
            assistHitFeedbackSource = feedbackRemote
            assistHitFeedbackConn = feedbackRemote.OnClientEvent:Connect(function(ok, message)
                if ok then
                    assistHitStatus = "Hit: " .. tostring(message)
                    if assistHitLastRequestedTarget and assistHitLastRequestedTarget.Parent == Players then
                        assistHitLockedTarget = assistHitLastRequestedTarget
                        assistHitLockExpiresAt = os.clock() + assistHitLockTime
                    end
                else
                    assistHitStatus = "Blocked: " .. tostring(message)
                end
            end)
        end
    end

    return tryAttackRemote, feedbackRemote
end

local function makeDraggable(frame)
    local dragging, dragStart, startPos

    frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and (
            input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch
        ) then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end)
end

local function isPlayerAlive(p)
    if not espHpCheck then
        return true
    end

    if not p.Character then
        return false
    end

    local hum = p.Character:FindFirstChildOfClass("Humanoid")
    if not hum then
        return false
    end

    return hum.Health > 0.1
end

local function isCombatTargetAlive(p)
    if not p.Character then
        return false
    end

    local hum = p.Character:FindFirstChildOfClass("Humanoid")
    local root = p.Character:FindFirstChild("HumanoidRootPart")
    if not hum or not root then
        return false
    end

    return hum.Health > 0
end

local function isSameTeam(a, b)
    if not assistHitTeamCheck then
        return false
    end

    if a.Neutral or b.Neutral then
        return false
    end

    return a.Team ~= nil and a.Team == b.Team
end

local function hasAssistHitLineOfSight(targetCharacter)
    if not character or not targetCharacter then
        return false
    end

    local targetRoot = targetCharacter:FindFirstChild("HumanoidRootPart")
    if not targetRoot then
        return false
    end

    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = { character, targetCharacter }

    local origin = camera.CFrame.Position
    local direction = targetRoot.Position - origin
    local result = Workspace:Raycast(origin, direction, params)
    return result == nil
end

local function getLockedAssistTarget()
    if not assistHitLockedTarget then
        return nil
    end

    if os.clock() > assistHitLockExpiresAt then
        assistHitLockedTarget = nil
        return nil
    end

    if not hrp or not isCombatTargetAlive(assistHitLockedTarget) then
        assistHitLockedTarget = nil
        return nil
    end

    if friendsList[assistHitLockedTarget.Name] or isSameTeam(player, assistHitLockedTarget) then
        assistHitLockedTarget = nil
        return nil
    end

    local targetCharacter = assistHitLockedTarget.Character
    local targetRoot = targetCharacter and targetCharacter:FindFirstChild("HumanoidRootPart")
    if not targetRoot then
        assistHitLockedTarget = nil
        return nil
    end

    local worldDistance = (targetRoot.Position - hrp.Position).Magnitude
    if worldDistance > assistHitLockDistance then
        assistHitLockedTarget = nil
        return nil
    end

    if not hasAssistHitLineOfSight(targetCharacter) then
        assistHitLockedTarget = nil
        return nil
    end

    return assistHitLockedTarget
end

local function getClosestAssistHitTarget()
    if not hrp then
        return nil
    end

    local lockedTarget = getLockedAssistTarget()
    if lockedTarget then
        return lockedTarget
    end

    local viewportCenter = camera.ViewportSize / 2
    local bestTarget = nil
    local bestScreenDistance = assistHitMaxFovPixels

    for _, otherPlayer in ipairs(Players:GetPlayers()) do
        if otherPlayer ~= player
            and not friendsList[otherPlayer.Name]
            and isCombatTargetAlive(otherPlayer)
            and not isSameTeam(player, otherPlayer) then
            local targetCharacter = otherPlayer.Character
            local targetRoot = targetCharacter and targetCharacter:FindFirstChild("HumanoidRootPart")

            if targetRoot then
                local worldDistance = (targetRoot.Position - hrp.Position).Magnitude
                if worldDistance <= assistHitMaxDistance and hasAssistHitLineOfSight(targetCharacter) then
                    local screenPoint, onScreen = camera:WorldToViewportPoint(targetRoot.Position)
                    if onScreen and screenPoint.Z > 0 then
                        local screenDistance = (Vector2.new(screenPoint.X, screenPoint.Y) - viewportCenter).Magnitude
                        if screenDistance < bestScreenDistance then
                            bestScreenDistance = screenDistance
                            bestTarget = otherPlayer
                        end
                    end
                end
            end
        end
    end

    return bestTarget
end

local function stopAssistHit()
    if assistHitInputConn then
        assistHitInputConn:Disconnect()
        assistHitInputConn = nil
    end

    assistHitLastRequestedTarget = nil
    assistHitLockedTarget = nil
    assistHitLockExpiresAt = 0
end

local function startAssistHit()
    stopAssistHit()

    if not assistHitEnabled then
        return
    end

    getAssistHitRemotes()

    assistHitInputConn = UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed or not assistHitEnabled then
            return
        end

        if input.UserInputType ~= Enum.UserInputType.MouseButton1 then
            return
        end

        local tryAttackRemote = getAssistHitRemotes()
        if not tryAttackRemote then
            assistHitStatus = "Missing GameRemotes"
            return
        end

        local target = getClosestAssistHitTarget()
        if not target then
            assistHitStatus = "No target"
            return
        end

        assistHitLastRequestedTarget = target
        assistHitStatus = "Request: " .. target.Name
        tryAttackRemote:FireServer(target)
    end)
end

local lastTargetNameCache = ""
local function updateTargetHud(targetPlayer)
    if not targetPlayer then
        targetHudFrame.Visible = false
        lastTargetNameCache = ""
        return
    end

    targetHudFrame.Visible = true
    targetNameLabel.Text = targetPlayer.Name

    if lastTargetNameCache ~= targetPlayer.Name then
        lastTargetNameCache = targetPlayer.Name
        pcall(function()
            local content = Players:GetUserThumbnailAsync(
                targetPlayer.UserId,
                Enum.ThumbnailType.HeadShot,
                Enum.ThumbnailSize.Size60x60
            )
            targetAvatar.Image = content
        end)
    end

    if targetPlayer.Character then
        local hum = targetPlayer.Character:FindFirstChildOfClass("Humanoid")
        if hum then
            local hp = math.floor(hum.Health)
            local maxHp = math.floor(hum.MaxHealth)
            local pct = math.clamp(hum.Health / math.max(hum.MaxHealth, 1), 0, 1)

            targetHPLabel.Text = string.format("HP: %d / %d", hp, maxHp)
            hpBarFill.Size = UDim2.new(pct, 0, 1, 0)

            if pct > 0.6 then
                hpBarFill.BackgroundColor3 = Color3.fromRGB(80, 200, 80)
                targetHPLabel.TextColor3 = Color3.fromRGB(100, 220, 100)
            elseif pct > 0.3 then
                hpBarFill.BackgroundColor3 = Color3.fromRGB(220, 180, 40)
                targetHPLabel.TextColor3 = Color3.fromRGB(220, 180, 40)
            else
                hpBarFill.BackgroundColor3 = Color3.fromRGB(220, 60, 60)
                targetHPLabel.TextColor3 = Color3.fromRGB(220, 60, 60)
            end
        end
    end
end

local function safeLoad(url)
    local ok, result = pcall(function()
        return loadstring(game:HttpGet(url))()
    end)

    if not ok then
        warn("Failed to load: " .. url .. "\n" .. tostring(result))
        return nil
    end

    return result
end

local function isVisible(targetPart)
    if not hrp or not targetPart then
        return false
    end

    local ray = Ray.new(hrp.Position, (targetPart.Position - hrp.Position).Unit * 1000)
    local hit = Workspace:FindPartOnRayWithIgnoreList(ray, { character, targetPart.Parent })
    return hit == nil or hit == targetPart
end

local function getClosestPlayer()
    local closest, closestDist = nil, math.huge
    local myRoot = getRoot(character)

    if not myRoot then
        return nil
    end

    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character then
            local theirRoot = getRoot(p.Character)
            if theirRoot then
                if friendsList[p.Name] then
                    continue
                end

                local dist = (theirRoot.Position - myRoot.Position).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closest = p
                end
            end
        end
    end

    return closest
end

local function getAimbotPart(targetChar)
    if aimbotTarget == "Head" then
        return targetChar:FindFirstChild("Head")
    end

    return targetChar:FindFirstChild("HumanoidRootPart")
end

local function isAimbotVisible(targetPart)
    if not aimbotVisibleCheck then
        return true
    end

    if not character or not targetPart then
        return false
    end

    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = { character, targetPart.Parent }

    local origin = camera.CFrame.Position
    local direction = targetPart.Position - origin
    local result = Workspace:Raycast(origin, direction, params)
    return result == nil
end

local function getClosestAimbotTarget()
    if not hrp then
        return nil, nil
    end

    local viewportCenter = camera.ViewportSize / 2
    local bestPlayer = nil
    local bestPart = nil
    local bestScreenDistance = aimbotFovPixels

    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and not friendsList[p.Name] and isCombatTargetAlive(p) then
            local part = getAimbotPart(p.Character)
            local root = getRoot(p.Character)

            if part and root then
                local worldDistance = (root.Position - hrp.Position).Magnitude
                if worldDistance <= aimbotMaxDistance and isAimbotVisible(part) then
                    local screenPoint, onScreen = camera:WorldToViewportPoint(part.Position)

                    if onScreen and screenPoint.Z > 0 then
                        local screenDistance = (Vector2.new(screenPoint.X, screenPoint.Y) - viewportCenter).Magnitude

                        if screenDistance < bestScreenDistance then
                            bestScreenDistance = screenDistance
                            bestPlayer = p
                            bestPart = part
                        end
                    end
                end
            end
        end
    end

    return bestPlayer, bestPart
end

                mouse1press()
                task.wait()
                mouse1release()
            end
        end
    end)

local aimbotLoop = nil

local function stopAimbot()
    if aimbotLoop then
        aimbotLoop:Disconnect()
        aimbotLoop = nil
    end

local function startAimbot()
    stopAimbot()

    if not aimbotEnabled then
        return
    end

    aimbotLoop = RunService.RenderStepped:Connect(function(deltaTime)
        if not aimbotEnabled then
            return
        end

        if not UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
            return
        end

        local target, part = getClosestAimbotTarget()
        if not target or not part then
            return
        end

        local predictedPosition = part.Position + (part.AssemblyLinearVelocity * aimbotPrediction)
        local targetCFrame = CFrame.new(camera.CFrame.Position, predictedPosition)
        local alpha = math.clamp(1 - math.exp(-aimbotSmoothness * deltaTime), 0, 1)

        camera.CFrame = camera.CFrame:Lerp(targetCFrame, alpha)
    end)
end

local function removeESP(p)
    if espObjects[p.Name] then
        for _, obj in pairs(espObjects[p.Name]) do
            pcall(function()
                if typeof(obj) == "RBXScriptConnection" then
                    obj:Disconnect()
                else
                    obj:Destroy()
                end
            end)
        end

        espObjects[p.Name] = nil
    end
end

local function buildESP(p, char)
    if not char or p == player then
        return
    end

    if not isPlayerAlive(p) then
        return
    end

    local isFriend = friendsList[p.Name]
    local col = isFriend and espFriendColor or espColor
    espObjects[p.Name] = {}

    if espMode == "Highlight" then
        local highlight = Instance.new("Highlight")
        highlight.Name = "ESPHighlight"
        highlight.Adornee = char
        highlight.FillColor = col
        highlight.OutlineColor = col
        highlight.FillTransparency = espHighlightFillTransparency
        highlight.OutlineTransparency = espHighlightOutlineTransparency
        highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        highlight.Parent = Workspace
        table.insert(espObjects[p.Name], highlight)

        local hpConn = RunService.Heartbeat:Connect(function()
            if not highlight or not highlight.Parent then
                return
            end

            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                highlight.Enabled = not (espHpCheck and hum.Health <= 0.1)
            end
        end)

        table.insert(espObjects[p.Name], hpConn)
    elseif espMode == "3D" then
        local box = Instance.new("SelectionBox")
        box.Name = "ESP3D"
        box.Color3 = col
        box.LineThickness = esp3DLineThickness
        box.SurfaceTransparency = 1
        box.SurfaceColor3 = col
        box.Adornee = char
        box.Parent = Workspace
        table.insert(espObjects[p.Name], box)

        local hpConn = RunService.Heartbeat:Connect(function()
            if not box or not box.Parent then
                return
            end

            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                if espHpCheck and hum.Health <= 0.1 then
                    box.Adornee = nil
                else
                    box.Adornee = char
                end
            end
        end)

        table.insert(espObjects[p.Name], hpConn)
    elseif espMode == "2D" then
        local hrpTarget = char:FindFirstChild("HumanoidRootPart")
        if not hrpTarget then
            hrpTarget = char:WaitForChild("HumanoidRootPart", 5)
            if not hrpTarget then
                return
            end
        end

        local bb = Instance.new("BillboardGui")
        bb.Name = "ESP2D"
        bb.Size = UDim2.new(0, 50, 0, 80)
        bb.AlwaysOnTop = true
        bb.Adornee = hrpTarget
        bb.Parent = hrpTarget

        local frame = Instance.new("Frame")
        frame.Size = UDim2.new(1, 0, 1, 0)
        frame.BackgroundTransparency = 1
        frame.BorderSizePixel = 0
        frame.Parent = bb

        local stroke = Instance.new("UIStroke")
        stroke.Color = col
        stroke.Thickness = 2
        stroke.Parent = frame

        local nameLabel = Instance.new("TextLabel")
        nameLabel.Size = UDim2.new(1, 0, 0, 14)
        nameLabel.Position = UDim2.new(0, 0, -0.22, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text = p.Name
        nameLabel.TextColor3 = col
        nameLabel.TextSize = 10
        nameLabel.Font = Enum.Font.GothamBold
        nameLabel.Parent = bb

        local hpBg = Instance.new("Frame")
        hpBg.Size = UDim2.new(0, 4, 1, 0)
        hpBg.Position = UDim2.new(0, -8, 0, 0)
        hpBg.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        hpBg.BorderSizePixel = 0
        hpBg.Parent = bb

        local hpFill = Instance.new("Frame")
        hpFill.Size = UDim2.new(1, 0, 1, 0)
        hpFill.AnchorPoint = Vector2.new(0, 1)
        hpFill.Position = UDim2.new(0, 0, 1, 0)
        hpFill.BackgroundColor3 = Color3.fromRGB(80, 200, 80)
        hpFill.BorderSizePixel = 0
        hpFill.Parent = hpBg

        local hpConn = RunService.Heartbeat:Connect(function()
            if not hpFill or not hpFill.Parent then
                return
            end

            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                if espHpCheck and hum.Health <= 0.1 then
                    bb.Enabled = false
                    return
                else
                    bb.Enabled = true
                end

                local pct = math.clamp(hum.Health / math.max(hum.MaxHealth, 1), 0, 1)
                hpFill.Size = UDim2.new(1, 0, pct, 0)

                if pct > 0.6 then
                    hpFill.BackgroundColor3 = Color3.fromRGB(80, 200, 80)
                elseif pct > 0.3 then
                    hpFill.BackgroundColor3 = Color3.fromRGB(220, 180, 40)
                else
                    hpFill.BackgroundColor3 = Color3.fromRGB(220, 60, 60)
                end
            end
        end)

        table.insert(espObjects[p.Name], bb)
        table.insert(espObjects[p.Name], hpConn)
    end
end

local function createESP(p)
    if p == player then
        return
    end

    if p.Character then
        buildESP(p, p.Character)
    end

    p.CharacterRemoving:Connect(function()
        removeESP(p)
    end)

    p.CharacterAdded:Connect(function(char)
        removeESP(p)
        task.wait(0.5)

        if espEnabled then
            buildESP(p, char)
        end
    end)
end

local function refreshESP()
    for _, p in ipairs(Players:GetPlayers()) do
        removeESP(p)
        if espEnabled then
            createESP(p)
        end
    end
end

local function removeNametag(p)
    local key = p.Name .. "_ntconn"

    if nametagObjects[key] then
        pcall(function()
            nametagObjects[key]:Disconnect()
        end)
        nametagObjects[key] = nil
    end

    if nametagObjects[p.Name] then
        pcall(function()
            nametagObjects[p.Name]:Destroy()
        end)
        nametagObjects[p.Name] = nil
    end
end

local function createNametag(p)
    if p == player then
        return
    end

    removeNametag(p)

    local function buildTag(char)
        if not char then
            return
        end

        local head = char:FindFirstChild("Head")
        if not head then
            return
        end

        if not isPlayerAlive(p) then
            return
        end

        local bb = Instance.new("BillboardGui")
        bb.Size = UDim2.new(0, 100, 0, 20)
        bb.StudsOffset = Vector3.new(0, 2.5, 0)
        bb.AlwaysOnTop = true
        bb.Adornee = head
        bb.Parent = head

        local lbl = Instance.new("TextLabel")
        lbl.Size = UDim2.new(1, 0, 1, 0)
        lbl.BackgroundTransparency = 1
        lbl.Text = p.Name
        lbl.TextColor3 = friendsList[p.Name] and espFriendColor or Color3.fromRGB(255, 255, 255)
        lbl.TextSize = 6
        lbl.Font = Enum.Font.GothamBold
        lbl.Parent = bb

        nametagObjects[p.Name] = bb

        local conn = RunService.Heartbeat:Connect(function()
            if not bb or not bb.Parent then
                return
            end

            local hum = char:FindFirstChildOfClass("Humanoid")

    end)

    p.CharacterAdded:Connect(function(char)
        task.wait(0.5)

    end)

local function isAirborne(hum)
    if not hum then
        return false
    end

    local state = hum:GetState()
    return hum.FloorMaterial == Enum.Material.Air and (
        state == Enum.HumanoidStateType.Freefall
        or state == Enum.HumanoidStateType.Jumping
        or state == Enum.HumanoidStateType.FallingDown
    )
end

local function stopNofall()
    nofallEnabled = false

    if nofallConn then
        nofallConn:Disconnect()
        nofallConn = nil
    end
end

local function startNofall()
    stopNofall()
    nofallEnabled = true

    nofallConn = RunService.Stepped:Connect(function(_, deltaTime)
        if not nofallEnabled or not character or not humanoid or not hrp then
            return
        end

        if not isAirborne(humanoid) then
            return
        end

        local velocity = hrp.AssemblyLinearVelocity
        if velocity.Y < -0.5 then
            local targetY = -glideFallSpeed
            if velocity.Y < targetY then
                local alpha = math.clamp(deltaTime * 10, 0.08, 0.35)
                local newY = velocity.Y + (targetY - velocity.Y) * alpha

                hrp.AssemblyLinearVelocity = Vector3.new(
                    velocity.X,
                    newY,
                    velocity.Z
                )
            end
        end
    end)
end

local function stopSwim()
    swimEnabled = false

    if swimConn then
        swimConn:Disconnect()
        swimConn = nil
    end

    if originalSwimGravity ~= nil then
        Workspace.Gravity = originalSwimGravity
        originalSwimGravity = nil
    end

    if humanoid then
        humanoid:SetStateEnabled(Enum.HumanoidStateType.GettingUp, true)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.Running, true)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.RunningNoPhysics, true)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.Freefall, true)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, true)
        humanoid:ChangeState(Enum.HumanoidStateType.RunningNoPhysics)

        if originalSwimWalkSpeed ~= nil then
            humanoid.WalkSpeed = originalSwimWalkSpeed
            originalSwimWalkSpeed = nil
        else
            humanoid.WalkSpeed = 16
        end
    end

    if hrp then
        hrp.AssemblyLinearVelocity = Vector3.zero
    end
end

startSwim = function()
    stopSwim()
    swimEnabled = true

    if not humanoid or not hrp then
        return
    end

    originalSwimGravity = Workspace.Gravity
    originalSwimWalkSpeed = humanoid.WalkSpeed
    Workspace.Gravity = 0

    humanoid:SetStateEnabled(Enum.HumanoidStateType.GettingUp, false)
    humanoid:SetStateEnabled(Enum.HumanoidStateType.Running, false)
    humanoid:SetStateEnabled(Enum.HumanoidStateType.RunningNoPhysics, false)
    humanoid:SetStateEnabled(Enum.HumanoidStateType.Freefall, false)
    humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
    humanoid:ChangeState(Enum.HumanoidStateType.Swimming)

    swimConn = RunService.RenderStepped:Connect(function()
        if not swimEnabled or not character or not humanoid or not hrp then
            return
        end

        humanoid.WalkSpeed = swimSpeed

        if humanoid:GetState() ~= Enum.HumanoidStateType.Swimming then
            humanoid:ChangeState(Enum.HumanoidStateType.Swimming)
        end

        local velocity = Vector3.zero
        local moveDirection = humanoid.MoveDirection

        if moveDirection.Magnitude > 0 then
            velocity += moveDirection.Unit * swimSpeed
        end

        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
            velocity += Vector3.new(0, swimSpeed, 0)
        end

        if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
            velocity += Vector3.new(0, -swimSpeed, 0)
        end

        hrp.AssemblyLinearVelocity = velocity
    end)
end

        speedConn:Disconnect()

        if hum then

local function startSpeed()
    stopSpeed()
    speedEnabled = true

    if speedMode == "Strafe" then
        speedConn = RunService.Heartbeat:Connect(function()
            if not speedEnabled or not character then
                return
            end

            local hum = character:FindFirstChildOfClass("Humanoid")
            if not hum then
                return
            end

            if hum.MoveDirection.Magnitude > 0 then
                hum.WalkSpeed = 16 * speedValue
            else
                hum.WalkSpeed = 16
            end
        end)
    elseif speedMode == "TpWalk" then
        speedConn = RunService.Heartbeat:Connect(function()
            if not speedEnabled or not character or not hrp then
                return
            end

            local hum = character:FindFirstChildOfClass("Humanoid")
            if not hum then
                return
            end

            local moveDir = hum.MoveDirection
            if moveDir.Magnitude > 0 then
                hrp.CFrame = hrp.CFrame + moveDir * speedValue
            end
        end)
    end
end
